#include <functional>
#include <iostream>

using namespace std;

int increase(int);
int apply(function<int(int)>, int);

int main(int argc, char **argv)
{
	cout << apply(increase, 9) << endl; // 10

	return 0;
}

int increase(int value)
{
	return value + 1;
}

int apply(function<int(int)> fx, int value)
{
	return fx(value);
}
